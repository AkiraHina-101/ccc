
'===============================================================================
' MODULE: ModTOCUpdate
' PURPOSE: Rebuild the Table of Contents (TOC) in the active PowerPoint file.
'
' HOW TO RUN:
'   1. Click inside the TOC textbox, or select the textbox border.
'   2. Run UpdateTOC from Alt+F8.
'
' WHAT IT DOES:
'   - Reads slide headers from slides after the TOC slide.
'   - Builds TOC lines with ruler/tab-stop alignment.
'   - Adds links from TOC page text to the first slide of each entry.
'   - Adds links from each slide header back to the TOC slide.
'
' DESIGN NOTE:
'   This module intentionally keeps helper names and comments explicit. A reader
'   should be able to see what each helper takes in, returns, and changes.
'===============================================================================

Public Sub UpdateTOCRibbon(control As IRibbonControl)
    UpdateTOC
End Sub

Public Sub UpdateTOC()
    Dim pres As Presentation
    Set pres = ActivePresentation

    Dim tocBox As Shape
    Set tocBox = GetSelectedTOCBox(pres.Windows(1).Selection)
    If tocBox Is Nothing Then
        MsgBox "Please click inside the TOC textbox or select its border, then run UpdateTOC again."
        Exit Sub
    End If
    If Not tocBox.HasTextFrame Then
        MsgBox "Selected shape has no text frame. Please select a textbox."
        Exit Sub
    End If

    Dim tocSlide As Slide
    Set tocSlide = tocBox.Parent

    Dim tocSlideIndex As Long
    tocSlideIndex = tocSlide.slideIndex
    If tocSlideIndex >= pres.Slides.count Then
        MsgBox "No slides found after the TOC slide. Nothing to scan."
        Exit Sub
    End If

    Dim oldParentKeys(50) As String
    Dim oldParentTitles(50) As String
    Dim oldParentCount As Long
    oldParentCount = 0
    ReadOldParentTitles tocBox, oldParentKeys, oldParentTitles, oldParentCount

    Dim entryKeys(200) As String
    Dim entryTitles(200) As String
    Dim entryFirstSlides(200) As Long
    Dim entryLastSlides(200) As Long
    Dim entryCount As Long
    entryCount = 0

    Dim slideIndex As Long
    For slideIndex = tocSlideIndex + 1 To pres.Slides.count
        Dim headerText As String
        headerText = GetHeaderText(pres.Slides(slideIndex))
        If Len(headerText) = 0 Then GoTo NextSlide

        Dim sectionKey As String
        sectionKey = ParseSectionKey(headerText)
        If Len(sectionKey) = 0 Then GoTo NextSlide

        Dim titleText As String
        titleText = ExtractTitle(headerText, sectionKey)

        Dim foundExistingEntry As Boolean
        foundExistingEntry = False

        Dim entryIndex As Long
        For entryIndex = 1 To entryCount
            If entryKeys(entryIndex) = sectionKey And entryTitles(entryIndex) = titleText Then
                entryLastSlides(entryIndex) = slideIndex
                foundExistingEntry = True
                Exit For
            End If
        Next entryIndex

        If Not foundExistingEntry Then
            entryCount = entryCount + 1
            entryKeys(entryCount) = sectionKey
            entryTitles(entryCount) = titleText
            entryFirstSlides(entryCount) = slideIndex
            entryLastSlides(entryCount) = slideIndex
        End If
NextSlide:
    Next slideIndex

    Dim maxSectionDepth As Long
    maxSectionDepth = 1
    For entryIndex = 1 To entryCount
        If SectionDepth(entryKeys(entryIndex)) > maxSectionDepth Then
            maxSectionDepth = SectionDepth(entryKeys(entryIndex))
        End If
    Next entryIndex

    Dim tocLines(200) As String
    Dim tocLineFirstSlide(200) As Long
    Dim tocLineCount As Long
    tocLineCount = 0

    Dim emittedParentKeys(50) As String
    Dim emittedParentCount As Long
    emittedParentCount = 0

    Dim pageWord As String
    pageWord = ChrW(12506) & ChrW(12540) & ChrW(12472)

    Dim arrow As String
    arrow = ChrW(8594)

    Dim tabChar As String
    tabChar = Chr(9)

    For entryIndex = 1 To entryCount
        Dim dotPosition As Long
        dotPosition = InStr(entryKeys(entryIndex), ".")

        If dotPosition > 0 Then
            Dim parentKey As String
            parentKey = Left(entryKeys(entryIndex), dotPosition - 1)

            If Not ParentAlreadyEmitted(parentKey, emittedParentKeys, emittedParentCount) Then
                Dim parentTitle As String
                parentTitle = LookupOldParentTitle(parentKey, oldParentKeys, oldParentTitles, oldParentCount)

                tocLineCount = tocLineCount + 1
                tocLineFirstSlide(tocLineCount) = 0
                If Len(parentTitle) > 0 Then
                    tocLines(tocLineCount) = parentKey & "." & tabChar & parentTitle
                Else
                    tocLines(tocLineCount) = parentKey & "."
                End If

                emittedParentCount = emittedParentCount + 1
                emittedParentKeys(emittedParentCount) = parentKey
            End If

            tocLineCount = tocLineCount + 1
            tocLineFirstSlide(tocLineCount) = entryFirstSlides(entryIndex)
            tocLines(tocLineCount) = FormatTOCLine( _
                entryKeys(entryIndex), entryTitles(entryIndex), maxSectionDepth, tabChar) & _
                arrow & " " & pageWord & " " & _
                PageRangeText(entryFirstSlides(entryIndex), entryLastSlides(entryIndex))

        Else
            If HasChildSections(entryKeys(entryIndex), entryKeys, entryCount) Then
                If Not ParentAlreadyEmitted(entryKeys(entryIndex), emittedParentKeys, emittedParentCount) Then
                    emittedParentCount = emittedParentCount + 1
                    emittedParentKeys(emittedParentCount) = entryKeys(entryIndex)
                End If
            End If

            tocLineCount = tocLineCount + 1
            tocLineFirstSlide(tocLineCount) = entryFirstSlides(entryIndex)
            tocLines(tocLineCount) = FormatTOCLine( _
                entryKeys(entryIndex), entryTitles(entryIndex), maxSectionDepth, tabChar) & _
                arrow & " " & pageWord & " " & _
                PageRangeText(entryFirstSlides(entryIndex), entryLastSlides(entryIndex))
        End If
    Next entryIndex

    If tocLineCount = 0 Then
        MsgBox "No TOC entries found. Check that content slides after the TOC slide contain a text shape named '" & _
            HEADER_SHAPE_NAME & "' and that its text starts with a numeric section key like 1, 1.1, or 1.1.1."
        Exit Sub
    End If

    Dim tocText As String
    tocText = ""

    Dim tocLineIndex As Long
    For tocLineIndex = 1 To tocLineCount
        If tocLineIndex > 1 Then tocText = tocText & Chr(13)
        tocText = tocText & tocLines(tocLineIndex)
    Next tocLineIndex

    tocBox.TextFrame.TextRange.Text = tocText
    ApplyTOCPageLinks tocBox, pres, tocLineFirstSlide, tocLineCount, pageWord
    NormalizeTOCTextColor tocBox
    LinkSlideHeadersBackToTOC pres, tocSlideIndex
    pres.Save

    MsgBox "TOC updated! (" & tocLineCount & " lines written)"
End Sub

'===============================================================================
' HELPER FUNCTIONS
'===============================================================================

' IN : Current PowerPoint selection.
' OUT: TOC textbox shape, or Nothing.
' NOTE: Works when the user selected the textbox border OR clicked inside text.
Private Function GetSelectedTOCBox(sel As Selection) As Shape
    On Error GoTo NoShape

    If sel.Type = ppSelectionShapes Then
        Set GetSelectedTOCBox = sel.ShapeRange(1)
        Exit Function
    End If

    If sel.Type = ppSelectionText Then
        Set GetSelectedTOCBox = sel.TextRange.Parent.Parent
        Exit Function
    End If

NoShape:
    Set GetSelectedTOCBox = Nothing
End Function

' IN : One slide.
' OUT: Text from the configured header shape; "" if not found.
Private Function GetHeaderText(sld As Slide) As String
    Dim shp As Shape
    For Each shp In sld.Shapes
        If shp.Name = HEADER_SHAPE_NAME Then
            If shp.HasTextFrame Then
                GetHeaderText = Trim(shp.TextFrame.TextRange.Text)
                Exit Function
            End If
        End If
    Next shp
End Function

' IN : Header text, e.g. "5.3. Regression Model".
' OUT: Numeric key at the start, e.g. "5.3"; "" if no numeric key exists.
Private Function ParseSectionKey(headerText As String) As String
    Dim result As String
    result = ""

    Dim pos As Long
    pos = 1
    Do While pos <= Len(headerText)
        Dim codePoint As Long
        codePoint = AscW(Mid(headerText, pos, 1))
        If codePoint = 32 Or codePoint = 9 Or codePoint = 12288 Then
            pos = pos + 1
        Else
            Exit Do
        End If
    Loop

    Dim scanPos As Long
    Dim ch As String
    For scanPos = pos To Len(headerText)
        ch = Mid(headerText, scanPos, 1)
        If ch >= "0" And ch <= "9" Then
            result = result & ch
        ElseIf ch = "." Then
            If Len(result) = 0 Then Exit For
            If Right(result, 1) = "." Then Exit For
            If scanPos < Len(headerText) Then
                If Mid(headerText, scanPos + 1, 1) >= "0" And Mid(headerText, scanPos + 1, 1) <= "9" Then
                    result = result & "."
                Else
                    Exit For
                End If
            Else
                Exit For
            End If
        Else
            Exit For
        End If
    Next scanPos

    ParseSectionKey = result
End Function

' IN : Header text and its parsed section key.
' OUT: Title after the key; leading separators and trailing PowerPoint CR removed.
Private Function ExtractTitle(headerText As String, sectionKey As String) As String
    Dim pos As Long
    pos = Len(sectionKey) + 1

    Do While pos <= Len(headerText)
        Dim codePoint As Long
        codePoint = AscW(Mid(headerText, pos, 1))
        If codePoint = 46 Or codePoint = 9 Or codePoint = 32 Or codePoint = 12288 Or codePoint = 12290 Then
            pos = pos + 1
        Else
            Exit Do
        End If
    Loop

    Dim result As String
    result = Mid(headerText, pos)

    Do While Len(result) > 0
        codePoint = AscW(Right(result, 1))
        If codePoint = 13 Or codePoint = 10 Or codePoint = 11 Then
            result = Left(result, Len(result) - 1)
        Else
            Exit Do
        End If
    Loop

    ExtractTitle = result
End Function

' IN : Section key, e.g. "5", "5.3", "5.3.1".
' OUT: Level number: 1, 2, 3, ...
Private Function SectionDepth(sectionKey As String) As Long
    Dim i As Long
    SectionDepth = 1
    For i = 1 To Len(sectionKey)
        If Mid(sectionKey, i, 1) = "." Then SectionDepth = SectionDepth + 1
    Next i
End Function

' IN : Section key/title plus max depth and tab character.
' OUT: TOC text before the arrow/page part.
'      Example level 2: <TAB>5.3<TAB>Title<TAB>
Private Function FormatTOCLine(sectionKey As String, titleText As String, maxDepth As Long, tabChar As String) As String
    Dim depth As Long
    depth = SectionDepth(sectionKey)

    Dim displayKey As String
    displayKey = sectionKey
    If depth = 1 Then displayKey = displayKey & "."

    FormatTOCLine = RepeatToken(tabChar, depth) & displayKey & "  " & titleText & _
        RepeatToken(tabChar, PageTabsAfterTitle(titleText, depth, maxDepth))
End Function

' IN : Title text, current section depth, deepest section depth in this TOC.
' OUT: Number of TAB characters needed after title to reach the page ruler.
' NOTE: Long titles may already pass an intermediate ruler, so they need fewer tabs.
Private Function PageTabsAfterTitle(titleText As String, depth As Long, maxDepth As Long) As Long
    Dim baseTabCount As Long
    baseTabCount = maxDepth - depth + 1

    Dim passedIntermediateStops As Long
    passedIntermediateStops = ApproxTextWidthUnits(titleText) \ 10
    If passedIntermediateStops > baseTabCount - 1 Then passedIntermediateStops = baseTabCount - 1

    PageTabsAfterTitle = baseTabCount - passedIntermediateStops
    If PageTabsAfterTitle < 1 Then PageTabsAfterTitle = 1
End Function

' IN : Any text.
' OUT: Rough width score; ASCII = 1, Japanese/full-width = 2.
Private Function ApproxTextWidthUnits(textValue As String) As Long
    Dim i As Long
    ApproxTextWidthUnits = 0
    For i = 1 To Len(textValue)
        If AscW(Mid(textValue, i, 1)) < 128 Then
            ApproxTextWidthUnits = ApproxTextWidthUnits + 1
        Else
            ApproxTextWidthUnits = ApproxTextWidthUnits + 2
        End If
    Next i
End Function

' IN : Token and count.
' OUT: Token repeated count times.
Private Function RepeatToken(token As String, count As Long) As String
    Dim i As Long
    RepeatToken = ""
    For i = 1 To count
        RepeatToken = RepeatToken & token
    Next i
End Function

' IN : First and last slide index.
' OUT: "7" or "7~9" using full-width wave dash.
Private Function PageRangeText(firstSlide As Long, lastSlide As Long) As String
    If lastSlide > firstSlide Then
        PageRangeText = CStr(firstSlide) & ChrW(65374) & CStr(lastSlide)
    Else
        PageRangeText = CStr(firstSlide)
    End If
End Function

' IN : TOC textbox, presentation, first slide per TOC line, line count, page word.
' OUT: None.
' SIDE EFFECT: Adds hyperlink only from "page word..." to the first slide.
Private Sub ApplyTOCPageLinks(tocBox As Shape, pres As Presentation, firstSlides() As Long, lineCount As Long, pageWord As String)
    Dim tr As TextRange
    Set tr = tocBox.TextFrame.TextRange

    Dim lineIndex As Long
    For lineIndex = 1 To lineCount
        If firstSlides(lineIndex) <= 0 Then GoTo NextLine
        If firstSlides(lineIndex) > pres.Slides.count Then GoTo NextLine

        Dim para As TextRange
        Set para = tr.Paragraphs(lineIndex)

        Dim lineText As String
        lineText = para.Text

        Dim pageStart As Long
        pageStart = InStr(lineText, pageWord)
        If pageStart = 0 Then GoTo NextLine

        Dim linkLength As Long
        linkLength = Len(lineText) - pageStart + 1
        Do While linkLength > 0
            Dim codePoint As Long
            codePoint = AscW(Mid(lineText, pageStart + linkLength - 1, 1))
            If codePoint = 13 Or codePoint = 10 Or codePoint = 11 Then
                linkLength = linkLength - 1
            Else
                Exit Do
            End If
        Loop
        If linkLength <= 0 Then GoTo NextLine

        Dim pageTextRange As TextRange
        Set pageTextRange = para.Characters(pageStart, linkLength)

        With pageTextRange.ActionSettings(ppMouseClick)
            .Action = ppActionHyperlink
            .Hyperlink.Address = ""
            .Hyperlink.SubAddress = SlideSubAddress(pres.Slides(firstSlides(lineIndex)))
        End With

        pageTextRange.Font.Color.RGB = RGB(0, 0, 0)
NextLine:
    Next lineIndex
End Sub

' IN : TOC textbox.
' OUT: None.
' SIDE EFFECT: Restores black text after PowerPoint applies hyperlink color.
Private Sub NormalizeTOCTextColor(tocBox As Shape)
    If Not tocBox.HasTextFrame Then Exit Sub
    If Not tocBox.TextFrame.HasText Then Exit Sub

    tocBox.TextFrame.TextRange.Font.Color.RGB = RGB(0, 0, 0)
End Sub

' IN : Presentation and TOC slide index.
' OUT: None.
' SIDE EFFECT: Adds a click link from each content-slide header back to the TOC.
Private Sub LinkSlideHeadersBackToTOC(pres As Presentation, tocSlideIndex As Long)
    Dim tocTarget As String
    tocTarget = SlideSubAddress(pres.Slides(tocSlideIndex))

    Dim slideIndex As Long
    For slideIndex = tocSlideIndex + 1 To pres.Slides.count
        Dim shp As Shape
        For Each shp In pres.Slides(slideIndex).Shapes
            If shp.Name = HEADER_SHAPE_NAME Then
                With shp.ActionSettings(ppMouseClick)
                    .Action = ppActionHyperlink
                    .Hyperlink.Address = ""
                    .Hyperlink.SubAddress = tocTarget
                End With
                Exit For
            End If
        Next shp
    Next slideIndex
End Sub

' IN : Slide object.
' OUT: PowerPoint internal hyperlink target for that slide.
Private Function SlideSubAddress(sld As Slide) As String
    SlideSubAddress = CStr(sld.SlideID) & "," & CStr(sld.slideIndex) & "," & sld.Name
End Function

' IN : Current TOC textbox.
' OUT: Fills parentKeys(1..parentCount) and parentTitles(1..parentCount).
' NOTE: Parent lines have no page word and no dot in the parsed key.
Private Sub ReadOldParentTitles(tocBox As Shape, parentKeys() As String, parentTitles() As String, parentCount As Long)
    Dim tr As TextRange
    Set tr = tocBox.TextFrame.TextRange

    Dim pageWord As String
    pageWord = ChrW(12506) & ChrW(12540) & ChrW(12472)

    Dim lineIndex As Long
    For lineIndex = 1 To tr.Paragraphs.count
        Dim lineText As String
        lineText = tr.Paragraphs(lineIndex).Text
        If InStr(lineText, pageWord) > 0 Then GoTo NextLine

        Dim sectionKey As String
        sectionKey = ParseSectionKey(lineText)
        If Len(sectionKey) = 0 Then GoTo NextLine
        If InStr(sectionKey, ".") > 0 Then GoTo NextLine

        parentCount = parentCount + 1
        parentKeys(parentCount) = sectionKey
        parentTitles(parentCount) = ExtractTitle(lineText, sectionKey)
NextLine:
    Next lineIndex
End Sub

' IN : Parent key and arrays filled by ReadOldParentTitles.
' OUT: Existing parent title, or "" if not found.
Private Function LookupOldParentTitle(parentKey As String, parentKeys() As String, parentTitles() As String, parentCount As Long) As String
    Dim i As Long
    For i = 1 To parentCount
        If parentKeys(i) = parentKey Then
            LookupOldParentTitle = parentTitles(i)
            Exit Function
        End If
    Next i
    LookupOldParentTitle = ""
End Function

' IN : Parent key and emitted parent array.
' OUT: True if that parent line was already written to the TOC.
Private Function ParentAlreadyEmitted(parentKey As String, emittedKeys() As String, emittedCount As Long) As Boolean
    Dim i As Long
    For i = 1 To emittedCount
        If emittedKeys(i) = parentKey Then
            ParentAlreadyEmitted = True
            Exit Function
        End If
    Next i
    ParentAlreadyEmitted = False
End Function

' IN : Parent key and all scanned entry keys.
' OUT: True if any entry key starts with "parentKey.".
Private Function HasChildSections(parentKey As String, entryKeys() As String, entryCount As Long) As Boolean
    Dim prefix As String
    prefix = parentKey & "."

    Dim i As Long
    For i = 1 To entryCount
        If Left(entryKeys(i), Len(prefix)) = prefix Then
            HasChildSections = True
            Exit Function
        End If
    Next i
    HasChildSections = False
End Function
