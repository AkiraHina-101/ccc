Private Const MASTER_TOTAL_SLIDE_SHAPE_NAME As String = "Slide Number"

Private Sub UpdateMasterTotalSlideCount(pres As Presentation)
    Dim shp As Shape

    On Error Resume Next
    Set shp = pres.SlideMaster.Shapes(MASTER_TOTAL_SLIDE_SHAPE_NAME)
    On Error GoTo 0

    If shp Is Nothing Then Exit Sub
    If Not shp.HasTextFrame Then Exit Sub

    shp.TextFrame.TextRange.Text = CStr(pres.Slides.Count)
End Sub
