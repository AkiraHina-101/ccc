Private Sub EnsureSlideNumbersVisible(pres As Presentation)
    Dim sld As Slide

    pres.PageSetup.FirstSlideNumber = 1

    For Each sld In pres.Slides
        sld.HeadersFooters.SlideNumber.Visible = msoTrue
    Next sld
End Sub
